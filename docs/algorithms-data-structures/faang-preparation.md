# ==================== 1. TWO SUM ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook, Apple
# –ß–∞—Å—Ç–æ—Ç–∞: –û—á–µ–Ω—å –≤—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Hash Map

def two_sum(nums, target):
    """
    –ù–∞–π—Ç–∏ –∏–Ω–¥–µ–∫—Å—ã –¥–≤—É—Ö —á–∏—Å–µ–ª, —Å—É–º–º–∞ –∫–æ—Ç–æ—Ä—ã—Ö —Ä–∞–≤–Ω–∞ target
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    """
    num_map = {}
    
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_map:
            return [num_map[complement], i]
        num_map[num] = i
    
    return []

# ==================== 2. VALID PARENTHESES ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook
# –ß–∞—Å—Ç–æ—Ç–∞: –û—á–µ–Ω—å –≤—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Stack

def is_valid_parentheses(s):
    """
    –ü—Ä–æ–≤–µ—Ä–∏—Ç—å –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ—Å—Ç—å —Å–∫–æ–±–æ–∫
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    """
    stack = []
    mapping = {')': '(', '}': '{', ']': '['}
    
    for char in s:
        if char in mapping:
            if not stack or stack.pop() != mapping[char]:
                return False
        else:
            stack.append(char)
    
    return len(stack) == 0

# ==================== 3. MERGE TWO SORTED LISTS ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook, Apple
# –ß–∞—Å—Ç–æ—Ç–∞: –í—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Two Pointers

class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def merge_two_lists(l1, l2):
    """
    –°–ª–∏—è–Ω–∏–µ –¥–≤—É—Ö –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å–ø–∏—Å–∫–æ–≤
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n + m)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
    """
    dummy = ListNode(0)
    current = dummy
    
    while l1 and l2:
        if l1.val <= l2.val:
            current.next = l1
            l1 = l1.next
        else:
            current.next = l2
            l2 = l2.next
        current = current.next
    
    # –î–æ–±–∞–≤–ª—è–µ–º –æ—Å—Ç–∞–≤—à–∏–µ—Å—è —ç–ª–µ–º–µ–Ω—Ç—ã
    current.next = l1 or l2
    
    return dummy.next
```

# ==================== 1. LONGEST SUBSTRING WITHOUT REPEATING ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook, Apple
# –ß–∞—Å—Ç–æ—Ç–∞: –û—á–µ–Ω—å –≤—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Sliding Window

def length_of_longest_substring(s):
    """
    –î–ª–∏–Ω–∞ —Å–∞–º–æ–π –¥–ª–∏–Ω–Ω–æ–π –ø–æ–¥—Å—Ç—Ä–æ–∫–∏ –±–µ–∑ –ø–æ–≤—Ç–æ—Ä–µ–Ω–∏–π
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(min(m, n))
    """
    char_set = set()
    left = 0
    max_length = 0
    
    for right in range(len(s)):
        while s[right] in char_set:
            char_set.remove(s[left])
            left += 1
        
        char_set.add(s[right])
        max_length = max(max_length, right - left + 1)
    
    return max_length

# ==================== 2. GROUP ANAGRAMS ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook
# –ß–∞—Å—Ç–æ—Ç–∞: –í—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Hash Map

def group_anagrams(strs):
    """
    –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ –∞–Ω–∞–≥—Ä–∞–º–º
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n * m log m)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n * m)
    """
    from collections import defaultdict
    
    anagram_groups = defaultdict(list)
    
    for s in strs:
        # –°–æ—Ä—Ç–∏—Ä—É–µ–º —Å–∏–º–≤–æ–ª—ã –∫–∞–∫ –∫–ª—é—á
        key = ''.join(sorted(s))
        anagram_groups[key].append(s)
    
    return list(anagram_groups.values())

# ==================== 3. MAXIMUM SUBARRAY ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Apple
# –ß–∞—Å—Ç–æ—Ç–∞: –û—á–µ–Ω—å –≤—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Kadane's Algorithm

def max_subarray(nums):
    """
    –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è —Å—É–º–º–∞ –ø–æ–¥–º–∞—Å—Å–∏–≤–∞ (Kadane's Algorithm)
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
    """
    max_ending_here = max_so_far = nums[0]
    
    for i in range(1, len(nums)):
        # –õ–∏–±–æ –ø—Ä–æ–¥–æ–ª–∂–∞–µ–º —Ç–µ–∫—É—â–∏–π –ø–æ–¥–º–∞—Å—Å–∏–≤, –ª–∏–±–æ –Ω–∞—á–∏–Ω–∞–µ–º –Ω–æ–≤—ã–π
        max_ending_here = max(nums[i], max_ending_here + nums[i])
        max_so_far = max(max_so_far, max_ending_here)
    
    return max_so_far
```

# ==================== 1. MEDIAN OF TWO SORTED ARRAYS ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Apple, Facebook
# –ß–∞—Å—Ç–æ—Ç–∞: –í—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Binary Search

def find_median_sorted_arrays(nums1, nums2):
    """
    –ú–µ–¥–∏–∞–Ω–∞ –¥–≤—É—Ö –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –º–∞—Å—Å–∏–≤–æ–≤
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(log(min(m, n)))
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
    """
    # –ì–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ–º, —á—Ç–æ nums1 –º–µ–Ω—å—à–µ nums2
    if len(nums1) > len(nums2):
        nums1, nums2 = nums2, nums1
    
    m, n = len(nums1), len(nums2)
    left, right = 0, m
    
    while left <= right:
        partition_x = (left + right) // 2
        partition_y = (m + n + 1) // 2 - partition_x
        
        # –≠–ª–µ–º–µ–Ω—Ç—ã —Å–ª–µ–≤–∞ –æ—Ç —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è
        max_left_x = float('-inf') if partition_x == 0 else nums1[partition_x - 1]
        max_left_y = float('-inf') if partition_y == 0 else nums2[partition_y - 1]
        
        # –≠–ª–µ–º–µ–Ω—Ç—ã —Å–ø—Ä–∞–≤–∞ –æ—Ç —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è
        min_right_x = float('inf') if partition_x == m else nums1[partition_x]
        min_right_y = float('inf') if partition_y == n else nums2[partition_y]
        
        if max_left_x <= min_right_y and max_left_y <= min_right_x:
            # –ù–∞–π–¥–µ–Ω–æ –ø—Ä–∞–≤–∏–ª—å–Ω–æ–µ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏–µ
            if (m + n) % 2 == 0:
                return (max(max_left_x, max_left_y) + min(min_right_x, min_right_y)) / 2
            else:
                return max(max_left_x, max_left_y)
        elif max_left_x > min_right_y:
            right = partition_x - 1
        else:
            left = partition_x + 1
    
    return -1

# ==================== 2. TRAPPING RAIN WATER ====================
# –ö–æ–º–ø–∞–Ω–∏–∏: Google, Amazon, Facebook
# –ß–∞—Å—Ç–æ—Ç–∞: –í—ã—Å–æ–∫–∞—è
# –ü–∞—Ç—Ç–µ—Ä–Ω: Two Pointers

def trap_rain_water(height):
    """
    –°–±–æ—Ä –¥–æ–∂–¥–µ–≤–æ–π –≤–æ–¥—ã
    
    –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
    –ü—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
    """
    if not height:
        return 0
    
    left = 0
    right = len(height) - 1
    left_max = right_max = 0
    water_trapped = 0
    
    while left < right:
        if height[left] < height[right]:
            if height[left] >= left_max:
                left_max = height[left]
            else:
                water_trapped += left_max - height[left]
            left += 1
        else:
            if height[right] >= right_max:
                right_max = height[right]
            else:
                water_trapped += right_max - height[right]
            right -= 1
    
    return water_trapped
```

# ==================== –¢–ò–ü–ò–ß–ù–ê–Ø –ó–ê–î–ê–ß–ê GOOGLE ====================
# –ü–æ–∏—Å–∫ –≤ 2D –º–∞—Ç—Ä–∏—Ü–µ

def search_matrix(matrix, target):
    """
    –ü–æ–∏—Å–∫ –≤ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–æ–π 2D –º–∞—Ç—Ä–∏—Ü–µ
    
    –ö–∞–∂–¥–∞—è —Å—Ç—Ä–æ–∫–∞ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–∞ —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ
    –ü–µ—Ä–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç –∫–∞–∂–¥–æ–π —Å—Ç—Ä–æ–∫–∏ –±–æ–ª—å—à–µ –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –ø—Ä–µ–¥—ã–¥—É—â–µ–π
    """
    if not matrix or not matrix[0]:
        return False
    
    rows, cols = len(matrix), len(matrix[0])
    left, right = 0, rows * cols - 1
    
    while left <= right:
        mid = (left + right) // 2
        mid_value = matrix[mid // cols][mid % cols]
        
        if mid_value == target:
            return True
        elif mid_value < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return False
```

    def validate(node, min_val, max_val):

            return True
        
        if node.val <= min_val or node.val >= max_val:
