```python
def resolution(clauses):
    """–ê–ª–≥–æ—Ä–∏—Ç–º —Ä–µ–∑–æ–ª—é—Ü–∏–∏"""
    new_clauses = set()
    
    while True:
        # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –≤—Å–µ—Ö –≤–æ–∑–º–æ–∂–Ω—ã—Ö —Ä–µ–∑–æ–ª—å–≤–µ–Ω—Ç
        for c1 in clauses:
            for c2 in clauses:
                if c1 != c2:
                    resolvent = resolve(c1, c2)
                    if resolvent is not None:
                        if resolvent == set():  # –ü—É—Å—Ç–æ–π –∫–ª–æ–∑
                            return "–ü—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∏–µ"
                        new_clauses.add(resolvent)
        
        if new_clauses.issubset(clauses):
            return "–í—ã–ø–æ–ª–Ω–∏–º–æ"
        
        clauses = clauses.union(new_clauses)

def resolve(clause1, clause2):
    """–†–µ–∑–æ–ª—é—Ü–∏—è –¥–≤—É—Ö –∫–ª–æ–∑–æ–≤"""
    for literal in clause1:
        if negate(literal) in clause2:
            # –ù–∞–π–¥–µ–Ω —Ä–µ–∑–æ–ª—é–∏—Ä—É–µ–º—ã–π –ª–∏—Ç–µ—Ä–∞–ª
            new_clause = (clause1 - {literal}) | (clause2 - {negate(literal)})
            return new_clause
    
    return None
```
