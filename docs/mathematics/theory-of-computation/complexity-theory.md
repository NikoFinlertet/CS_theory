    def demonstrate_sat(self):
        """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –∑–∞–¥–∞—á–∏ SAT"""
        
        def evaluate_cnf(formula, assignment):
            """–ü—Ä–æ–≤–µ—Ä–∫–∞ –≤—ã–ø–æ–ª–Ω–∏–º–æ—Å—Ç–∏ –ö–ù–§ —Ñ–æ—Ä–º—É–ª—ã"""
            for clause in formula:
                clause_satisfied = False
                for literal in clause:
                    if literal > 0:  # –ü–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–π –ª–∏—Ç–µ—Ä–∞–ª
                        if assignment.get(literal, False):
                            clause_satisfied = True
                            break
                    else:  # –û—Ç—Ä–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–π –ª–∏—Ç–µ—Ä–∞–ª
                        if not assignment.get(-literal, False):
                            clause_satisfied = True
                            break
                
                if not clause_satisfied:
                    return False
            return True
        
        def solve_sat_bruteforce(formula, variables):
            """–†–µ—à–µ–Ω–∏–µ SAT –ø–µ—Ä–µ–±–æ—Ä–æ–º (—ç–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–µ –≤—Ä–µ–º—è)"""
            n = len(variables)
            
            for i in range(2**n):
                assignment = {}
                for j, var in enumerate(variables):
                    assignment[var] = bool(i & (1 << j))
                
                if evaluate_cnf(formula, assignment):
                    return True, assignment
            
            return False, None
        
        # –ü—Ä–∏–º–µ—Ä: (x‚ÇÅ ‚à® ¬¨x‚ÇÇ ‚à® x‚ÇÉ) ‚àß (¬¨x‚ÇÅ ‚à® x‚ÇÇ) ‚àß (¬¨x‚ÇÇ ‚à® ¬¨x‚ÇÉ)
        formula = [
            [1, -2, 3],    # x‚ÇÅ ‚à® ¬¨x‚ÇÇ ‚à® x‚ÇÉ
            [-1, 2],       # ¬¨x‚ÇÅ ‚à® x‚ÇÇ  
            [-2, -3]       # ¬¨x‚ÇÇ ‚à® ¬¨x‚ÇÉ
        ]
        variables = [1, 2, 3]
        
        satisfiable, solution = solve_sat_bruteforce(formula, variables)
        
        print("=== –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è SAT ===")

    def demonstrate_clique(self):

        def has_clique(graph, k):
            """–ü—Ä–æ–≤–µ—Ä–∫–∞ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è –∫–ª–∏–∫–∏ —Ä–∞–∑–º–µ—Ä–∞ k"""

            vertices = list(graph.keys())
            
            # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞ —Ä–∞–∑–º–µ—Ä–∞ k
            for subset in combinations(vertices, k):
                is_clique = True
                
                # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –≤—Å–µ –ø–∞—Ä—ã –≤–µ—Ä—à–∏–Ω —Å–æ–µ–¥–∏–Ω–µ–Ω—ã
                for i in range(len(subset)):
                    for j in range(i + 1, len(subset)):
                        if subset[j] not in graph[subset[i]]:
                            is_clique = False
                            break
                    if not is_clique:
                        break
                
                if is_clique:
                    return True, subset
            
            return False, None
        
        # –ü—Ä–∏–º–µ—Ä –≥—Ä–∞—Ñ–∞
        graph = {
            'A': ['B', 'C', 'D'],
            'B': ['A', 'C'],
            'C': ['A', 'B', 'D'],
            'D': ['A', 'C'],
            'E': ['F'],

        }

            if has_k_clique:
                print(f"  –ù–∞–π–¥–µ–Ω–Ω–∞—è –∫–ª–∏–∫–∞: {clique}")
        
        return graph

class PolynomialReductions:
    """–î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è –ø–æ–ª–∏–Ω–æ–º–∏–∞–ª—å–Ω—ã—Ö —Å–≤–µ–¥–µ–Ω–∏–π –º–µ–∂–¥—É –∑–∞–¥–∞—á–∞–º–∏"""
    
    def sat_to_3sat(self, formula):
        """–°–≤–µ–¥–µ–Ω–∏–µ SAT –∫ 3-SAT"""
        
        result_formula = []
        new_var_counter = max(abs(lit) for clause in formula for lit in clause) + 1
        
        for clause in formula:
            if len(clause) <= 3:
                # –ö–ª–æ–∑ —É–∂–µ –ø–æ–¥—Ö–æ–¥—è—â–µ–π –¥–ª–∏–Ω—ã
                result_formula.append(clause)
            
            elif len(clause) == 4:
                # (a ‚à® b ‚à® c ‚à® d) ‚Üí (a ‚à® b ‚à® y) ‚àß (¬¨y ‚à® c ‚à® d)
                y = new_var_counter
                new_var_counter += 1
                
                result_formula.append([clause[0], clause[1], y])
                result_formula.append([-y, clause[2], clause[3]])
            
            else:
                # –î–ª—è –±–æ–ª–µ–µ –¥–ª–∏–Ω–Ω—ã—Ö –∫–ª–æ–∑–æ–≤: —Ü–µ–ø–æ—á–∫–∞ –≤—Å–ø–æ–º–æ–≥–∞—Ç–µ–ª—å–Ω—ã—Ö –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö
                # (a‚ÇÅ ‚à® a‚ÇÇ ‚à® ... ‚à® a‚Çô) ‚Üí 
                # (a‚ÇÅ ‚à® a‚ÇÇ ‚à® y‚ÇÅ) ‚àß (¬¨y‚ÇÅ ‚à® a‚ÇÉ ‚à® y‚ÇÇ) ‚àß ... ‚àß (¬¨y‚Çô‚Çã‚ÇÉ ‚à® a‚Çô‚Çã‚ÇÅ ‚à® a‚Çô)
                
                prev_var = clause[0]
                for i in range(1, len(clause) - 2):
                    y = new_var_counter
                    new_var_counter += 1
                    
                    if i == 1:
                        result_formula.append([prev_var, clause[i], y])
                    else:
                        result_formula.append([-prev_var, clause[i], y])
                    
                    prev_var = y
                
                # –ü–æ—Å–ª–µ–¥–Ω–∏–π –∫–ª–æ–∑
                result_formula.append([-prev_var, clause[-2], clause[-1]])
        
        return result_formula
    
    def sat_to_clique(self, formula):
        """–°–≤–µ–¥–µ–Ω–∏–µ 3-SAT –∫ –∑–∞–¥–∞—á–µ –æ –∫–ª–∏–∫–µ"""
        
        # –°–æ–∑–¥–∞–µ–º –≥—Ä–∞—Ñ –¥–ª—è –∑–∞–¥–∞—á–∏ –æ –∫–ª–∏–∫–µ
        vertices = []
        edges = []
        
        # –î–ª—è –∫–∞–∂–¥–æ–≥–æ –∫–ª–æ–∑–∞ —Å–æ–∑–¥–∞–µ–º –≤–µ—Ä—à–∏–Ω—ã (–ø–æ –æ–¥–Ω–æ–π –Ω–∞ –∫–∞–∂–¥—ã–π –ª–∏—Ç–µ—Ä–∞–ª)
        for i, clause in enumerate(formula):
            for j, literal in enumerate(clause):
                vertex_name = f"c{i}_l{j}_{literal}"
                vertices.append(vertex_name)
        
        # –°–æ–∑–¥–∞–µ–º —Ä–µ–±—Ä–∞ –º–µ–∂–¥—É –≤–µ—Ä—à–∏–Ω–∞–º–∏ –∏–∑ —Ä–∞–∑–Ω—ã—Ö –∫–ª–æ–∑–æ–≤,
        # –µ—Å–ª–∏ –æ–Ω–∏ –Ω–µ –ø—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∞—Ç –¥—Ä—É–≥ –¥—Ä—É–≥—É
        for i in range(len(vertices)):
            for j in range(i + 1, len(vertices)):
                v1_parts = vertices[i].split('_')
                v2_parts = vertices[j].split('_')
                
                clause1 = int(v1_parts[0][1:])  # –ù–æ–º–µ—Ä –∫–ª–æ–∑–∞
                clause2 = int(v2_parts[0][1:])
                literal1 = int(v1_parts[2])     # –õ–∏—Ç–µ—Ä–∞–ª
                literal2 = int(v2_parts[2])
                
                # –†–µ–±—Ä–æ –µ—Å—Ç—å, –µ—Å–ª–∏:
                # 1. –í–µ—Ä—à–∏–Ω—ã –∏–∑ —Ä–∞–∑–Ω—ã—Ö –∫–ª–æ–∑–æ–≤
                # 2. –õ–∏—Ç–µ—Ä–∞–ª—ã –Ω–µ –ø—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∞—Ç (–Ω–µ x –∏ ¬¨x)
                if clause1 != clause2 and literal1 != -literal2:
                    edges.append((vertices[i], vertices[j]))
        
        clique_size = len(formula)  # –†–∞–∑–º–µ—Ä –∫–ª–∏–∫–∏ = –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–ª–æ–∑–æ–≤
        
        return {
            'vertices': vertices,
            'edges': edges,
            'clique_size': clique_size,
            'explanation': '–ö–ª–∏–∫–∞ —Ä–∞–∑–º–µ—Ä–∞ k —Å—É—â–µ—Å—Ç–≤—É–µ—Ç ‚ü∫ –∏—Å—Ö–æ–¥–Ω–∞—è —Ñ–æ—Ä–º—É–ª–∞ –≤—ã–ø–æ–ª–Ω–∏–º–∞'
        }
    
    def clique_to_vertex_cover(self, graph, clique_size):
        """–°–≤–µ–¥–µ–Ω–∏–µ CLIQUE –∫ VERTEX-COVER"""
        
        # –°—Ç—Ä–æ–∏–º –¥–æ–ø–æ–ª–Ω–µ–Ω–∏–µ –≥—Ä–∞—Ñ–∞
        all_vertices = set(graph.keys())
        complement_edges = []
        
        for v1 in all_vertices:
            for v2 in all_vertices:
                if v1 != v2 and v2 not in graph.get(v1, []):
                    complement_edges.append((v1, v2))
        
        # –í –¥–æ–ø–æ–ª–Ω–µ–Ω–∏–∏ –≥—Ä–∞—Ñ–∞ –∏—â–µ–º –≤–µ—Ä—à–∏–Ω–Ω–æ–µ –ø–æ–∫—Ä—ã—Ç–∏–µ —Ä–∞–∑–º–µ—Ä–∞ n - k
        vertex_cover_size = len(all_vertices) - clique_size
        
        return {
            'complement_graph': complement_edges,
            'vertex_cover_size': vertex_cover_size,
            'explanation': 'G –∏–º–µ–µ—Ç –∫–ª–∏–∫—É —Ä–∞–∑–º–µ—Ä–∞ k ‚ü∫ ·∏† –∏–º–µ–µ—Ç –ø–æ–∫—Ä—ã—Ç–∏–µ —Ä–∞–∑–º–µ—Ä–∞ n-k'
        }

# –î–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏—è —Å–≤–µ–¥–µ–Ω–∏–π

# –ü—Ä–∏–º–µ—Ä —Å–≤–µ–¥–µ–Ω–∏—è SAT ‚Üí 3-SAT
