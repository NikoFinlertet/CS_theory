    end_rgb = hex_to_rgb(end_color)

    for i in range(steps):
        ratio = i / (steps - 1)
        r = int(start_rgb[0] + (end_rgb[0] - start_rgb[0]) * ratio)

        return len(self.vertices.get(vertex, []))
    
    def find_path(self, start, end, path=[]):
        """–ü–æ–∏—Å–∫ –ø—É—Ç–∏ –º–µ–∂–¥—É –≤–µ—Ä—à–∏–Ω–∞–º–∏"""
        path = path + [start]
        if start == end:
            return path
        
        for neighbor in self.vertices.get(start, []):
            if neighbor not in path:
                new_path = self.find_path(neighbor, end, path)
                if new_path:
                    return new_path
        return None
    
    def find_shortest_path(self, start, end):
        """–ü–æ–∏—Å–∫ –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –ø—É—Ç–∏ (BFS)"""
        from collections import deque
        
        queue = deque([(start, [start])])
        visited = set([start])
        
        while queue:
            vertex, path = queue.popleft()
            
            if vertex == end:
                return path
            
            for neighbor in self.vertices.get(vertex, []):
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, path + [neighbor]))
        
        return None

social_network.add_edge("Bob", "David")

import heapq

class WeightedGraph:
    def __init__(self):
        self.vertices = {}
    
    def add_edge(self, from_vertex, to_vertex, weight):
        if from_vertex not in self.vertices:
            self.vertices[from_vertex] = []
        self.vertices[from_vertex].append((to_vertex, weight))
    
    def dijkstra(self, start):
        """–ê–ª–≥–æ—Ä–∏—Ç–º –î–µ–π–∫—Å—Ç—Ä—ã –¥–ª—è –ø–æ–∏—Å–∫–∞ –∫—Ä–∞—Ç—á–∞–π—à–∏—Ö –ø—É—Ç–µ–π"""
        distances = {vertex: float('inf') for vertex in self.vertices}
        distances[start] = 0
        pq = [(0, start)]
        visited = set()
        
        while pq:
            current_distance, current_vertex = heapq.heappop(pq)
            
            if current_vertex in visited:
                continue
            
            visited.add(current_vertex)
            
            for neighbor, weight in self.vertices.get(current_vertex, []):
                distance = current_distance + weight
                
                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    heapq.heappush(pq, (distance, neighbor))
        
        return distances

import math

class Vector2D:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __add__(self, other):
        return Vector2D(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):
        return Vector2D(self.x - other.x, self.y - other.y)
    
    def __mul__(self, scalar):
        return Vector2D(self.x * scalar, self.y * scalar)
    
    def length(self):
        return math.sqrt(self.x ** 2 + self.y ** 2)
    
    def normalize(self):
        """–ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –≤–µ–∫—Ç–æ—Ä–∞ (–µ–¥–∏–Ω–∏—á–Ω—ã–π –≤–µ–∫—Ç–æ—Ä)"""
        length = self.length()
        if length == 0:
            return Vector2D(0, 0)
        return Vector2D(self.x / length, self.y / length)
    
    def dot(self, other):
        """–°–∫–∞–ª—è—Ä–Ω–æ–µ –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ"""
        return self.x * other.x + self.y * other.y
    
    def angle_to(self, other):
        """–£–≥–æ–ª –º–µ–∂–¥—É –≤–µ–∫—Ç–æ—Ä–∞–º–∏"""
        dot_product = self.dot(other)
        lengths = self.length() * other.length()
        if lengths == 0:
            return 0
        return math.acos(dot_product / lengths)
    
    def __repr__(self):
        return f"Vector2D({self.x}, {self.y})"

# –ò–≥—Ä–æ–≤–∞—è —Ñ–∏–∑–∏–∫–∞ - –¥–≤–∏–∂–µ–Ω–∏–µ –ø–µ—Ä—Å–æ–Ω–∞–∂–∞
class Player:

        self.velocity = velocity
    
    def update(self, dt):
        """–û–±–Ω–æ–≤–ª–µ–Ω–∏–µ –ø–æ–∑–∏—Ü–∏–∏"""
        self.position = self.position + self.velocity * dt
    
    def move_towards(self, target, speed):
        """–î–≤–∏–∂–µ–Ω–∏–µ –∫ —Ü–µ–ª–∏"""
        direction = (target - self.position).normalize()
        self.velocity = direction * speed

class Matrix2D:
    def __init__(self, values):
        self.values = values  # 2x2 –º–∞—Ç—Ä–∏—Ü–∞ –∫–∞–∫ —Å–ø–∏—Å–æ–∫ —Å–ø–∏—Å–∫–æ–≤

        """–£–º–Ω–æ–∂–µ–Ω–∏–µ –º–∞—Ç—Ä–∏—Ü"""
        if isinstance(other, Matrix2D):
            result = [[0, 0], [0, 0]]
            for i in range(2):
                for j in range(2):
                    for k in range(2):
                        result[i][j] += self.values[i][k] * other.values[k][j]
            return Matrix2D(result)
        elif isinstance(other, Vector2D):
            # –£–º–Ω–æ–∂–µ–Ω–∏–µ –º–∞—Ç—Ä–∏—Ü—ã –Ω–∞ –≤–µ–∫—Ç–æ—Ä
            x = self.values[0][0] * other.x + self.values[0][1] * other.y
            y = self.values[1][0] * other.x + self.values[1][1] * other.y
            return Vector2D(x, y)

        """–ú–∞—Ç—Ä–∏—Ü–∞ –ø–æ–≤–æ—Ä–æ—Ç–∞"""

        sin_a = math.sin(angle)

            [cos_a, -sin_a],

        """–°–∏–Ω—É—Å–æ–∏–¥–∞–ª—å–Ω–∞—è –∞–Ω–∏–º–∞—Ü–∏—è"""

import numpy as np

def solve_load_balancing(servers, requests):
    """
    –†–µ—à–µ–Ω–∏–µ –∑–∞–¥–∞—á–∏ –±–∞–ª–∞–Ω—Å–∏—Ä–æ–≤–∫–∏ –Ω–∞–≥—Ä—É–∑–∫–∏
    servers: [(capacity, cost_per_request), ...]
    requests: [request_count_per_hour, ...]
    """
    n_servers = len(servers)
    n_hours = len(requests)
    
    # –°–æ–∑–¥–∞–µ–º –º–∞—Ç—Ä–∏—Ü—É –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–π
    # x[i][j] = –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–ø—Ä–æ—Å–æ–≤ –Ω–∞ —Å–µ—Ä–≤–µ—Ä i –≤ —á–∞—Å j
    
    # –û–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è –ø–æ –º–æ—â–Ω–æ—Å—Ç–∏ —Å–µ—Ä–≤–µ—Ä–æ–≤
    A = []
    b = []
    
    for i, (capacity, _) in enumerate(servers):
        for j in range(n_hours):
            constraint = [0] * (n_servers * n_hours)
            constraint[i * n_hours + j] = 1
            A.append(constraint)
            b.append(capacity)
    
    # –û–≥—Ä–∞–Ω–∏—á–µ–Ω–∏—è –ø–æ –æ–±—Å–ª—É–∂–∏–≤–∞–Ω–∏—é –≤—Å–µ—Ö –∑–∞–ø—Ä–æ—Å–æ–≤
    for j in range(n_hours):
        constraint = [0] * (n_servers * n_hours)
        for i in range(n_servers):
            constraint[i * n_hours + j] = 1
        A.append(constraint)
        b.append(requests[j])

    x = np.linalg.lstsq(A, b, rcond=None)[0]
    
    return x.reshape(n_servers, n_hours)

def newton_method(f, df, x0, tolerance=1e-10, max_iterations=100):
    """
    –ú–µ—Ç–æ–¥ –ù—å—é—Ç–æ–Ω–∞ –¥–ª—è –ø–æ–∏—Å–∫–∞ –∫–æ—Ä–Ω–µ–π —Ñ—É–Ω–∫—Ü–∏–∏
    f: —Ñ—É–Ω–∫—Ü–∏—è
    df: –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏–∏
    x0: –Ω–∞—á–∞–ª—å–Ω–æ–µ –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ
    """
    x = x0
    for i in range(max_iterations):
        fx = f(x)
        if abs(fx) < tolerance:
            return x, i
        
        dfx = df(x)
        if abs(dfx) < 1e-15:
            raise ValueError("–ü—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è –±–ª–∏–∑–∫–∞ –∫ –Ω—É–ª—é")
        
        x = x - fx / dfx
    
    raise ValueError("–ù–µ —É–¥–∞–ª–æ—Å—å –Ω–∞–π—Ç–∏ –∫–æ—Ä–µ–Ω—å –∑–∞ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π")

# –ü—Ä–∏–º–µ—Ä: –ø–æ–∏—Å–∫ –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–≥–æ –∫–æ—Ä–Ω—è
def sqrt_newton(a, x0=None):
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–≥–æ –∫–æ—Ä–Ω—è –º–µ—Ç–æ–¥–æ–º –ù—å—é—Ç–æ–Ω–∞"""
    if x0 is None:
        x0 = a / 2
    
    f = lambda x: x**2 - a
    df = lambda x: 2*x
    
    root, iterations = newton_method(f, df, x0)
    return root, iterations

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ

sqrt_result, iterations = sqrt_newton(number)

print(f"–ü—Ä–æ–≤–µ—Ä–∫–∞: {sqrt_result**2:.10f}")

def trapezoidal_rule(f, a, b, n):
    """
    –ß–∏—Å–ª–µ–Ω–Ω–æ–µ –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏–µ –º–µ—Ç–æ–¥–æ–º —Ç—Ä–∞–ø–µ—Ü–∏–π
    f: —Ñ—É–Ω–∫—Ü–∏—è
    a, b: –ø—Ä–µ–¥–µ–ª—ã –∏–Ω—Ç–µ–≥—Ä–∏—Ä–æ–≤–∞–Ω–∏—è
    n: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ä–∞–∑–±–∏–µ–Ω–∏–π
    """
    h = (b - a) / n
    result = 0.5 * (f(a) + f(b))
    
    for i in range(1, n):
        x = a + i * h
        result += f(x)
    
    return result * h

# –ü—Ä–∏–º–µ—Ä: –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ –ø–ª–æ—â–∞–¥–∏ –ø–æ–¥ –∫—Ä–∏–≤–æ–π

    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ –ø–ª–æ—â–∞–¥–∏ –ø–æ–¥ –∫—Ä–∏–≤–æ–π –ø–æ —Ç–æ—á–∫–∞–º –¥–∞–Ω–Ω—ã—Ö"""
    def interpolate(x):
        # –ü—Ä–æ—Å—Ç–∞—è –ª–∏–Ω–µ–π–Ω–∞—è –∏–Ω—Ç–µ—Ä–ø–æ–ª—è—Ü–∏—è
        for i in range(len(data_points) - 1):
            x1, y1 = data_points[i]
            x2, y2 = data_points[i + 1]
            if x1 <= x <= x2:
                return y1 + (y2 - y1) * (x - x1) / (x2 - x1)
        return 0
    
    if len(data_points) < 2:
        return 0
    
    a = data_points[0][0]
    b = data_points[-1][0]
    
    return trapezoidal_rule(interpolate, a, b, 1000)

def simple_cipher(text, key):
    """–ü—Ä–æ—Å—Ç–æ–π —à–∏—Ñ—Ä –¶–µ–∑–∞—Ä—è"""
    result = ""
    for char in text:
        if char.isalpha():
            ascii_offset = ord('A') if char.isupper() else ord('a')
            shifted = (ord(char) - ascii_offset + key) % 26
            result += chr(shifted + ascii_offset)
        else:
            result += char
    return result

def gcd(a, b):
    """–ù–∞–∏–±–æ–ª—å—à–∏–π –æ–±—â–∏–π –¥–µ–ª–∏—Ç–µ–ª—å (–∞–ª–≥–æ—Ä–∏—Ç–º –ï–≤–∫–ª–∏–¥–∞)"""
    while b:
        a, b = b, a % b
    return a

def mod_inverse(a, m):
    """–ú–æ–¥—É–ª—å–Ω–æ–µ –æ–±—Ä–∞—â–µ–Ω–∏–µ"""
    if gcd(a, m) != 1:
        return None
    
    # –†–∞—Å—à–∏—Ä–µ–Ω–Ω—ã–π –∞–ª–≥–æ—Ä–∏—Ç–º –ï–≤–∫–ª–∏–¥–∞
    def extended_gcd(a, b):
        if a == 0:
            return b, 0, 1
        gcd, x1, y1 = extended_gcd(b % a, a)
        x = y1 - (b // a) * x1
        y = x1
        return gcd, x, y
    
    _, x, _ = extended_gcd(a, m)
    return (x % m + m) % m

# –ü—Ä–æ—Å—Ç–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è RSA (–¥–ª—è –¥–µ–º–æ–Ω—Å—Ç—Ä–∞—Ü–∏–∏)
def generate_rsa_keys(p, q):
    """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è –∫–ª—é—á–µ–π RSA"""
    n = p * q
    phi = (p - 1) * (q - 1)
    
    # –í—ã–±–∏—Ä–∞–µ–º e (–æ–±—ã—á–Ω–æ 65537)
    e = 65537
    if gcd(e, phi) != 1:
        e = 3
    
    # –í—ã—á–∏—Å–ª—è–µ–º d
    d = mod_inverse(e, phi)
    
    return (e, n), (d, n)  # (–æ—Ç–∫—Ä—ã—Ç—ã–π –∫–ª—é—á, –∑–∞–∫—Ä—ã—Ç—ã–π –∫–ª—é—á)

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
p, q = 61, 53  # –ü—Ä–æ—Å—Ç—ã–µ —á–∏—Å–ª–∞

class Vector3D:

        self.x = x
        self.y = y

        return Vector3D(self.x + other.x, self.y + other.y, self.z + other.z)

        return Vector3D(self.x - other.x, self.y - other.y, self.z - other.z)

        return self.x * other.x + self.y * other.y + self.z * other.z

            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x
        )
