import statistics

def measure_performance(func, *args, runs=100):

    for _ in range(runs):

        func(*args)

        times.append(end - start)
    
    return {
        'mean': statistics.mean(times),
        'median': statistics.median(times),
        'stdev': statistics.stdev(times),

        'max': max(times)
    }
