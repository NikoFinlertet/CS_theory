class BitSetOperations:
    """–û–ø–µ—Ä–∞—Ü–∏–∏ —Å –º–Ω–æ–∂–µ—Å—Ç–≤–∞–º–∏ —á–µ—Ä–µ–∑ –±–∏—Ç–æ–≤—ã–µ –º–∞—Å–∫–∏"""
    
    def __init__(self, max_size=32):
        self.bits = 0
        self.max_size = max_size
    
    def add(self, element):
        """–î–æ–±–∞–≤–∏—Ç—å —ç–ª–µ–º–µ–Ω—Ç –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–æ"""
        if 0 <= element < self.max_size:
            self.bits |= (1 << element)
    
    def remove(self, element):
        """–£–¥–∞–ª–∏—Ç—å —ç–ª–µ–º–µ–Ω—Ç –∏–∑ –º–Ω–æ–∂–µ—Å—Ç–≤–∞"""
        if 0 <= element < self.max_size:
            self.bits &= ~(1 << element)
    
    def contains(self, element):
        """–ü—Ä–æ–≤–µ—Ä–∏—Ç—å –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–Ω–æ—Å—Ç—å —ç–ª–µ–º–µ–Ω—Ç–∞ –º–Ω–æ–∂–µ—Å—Ç–≤—É"""
        if 0 <= element < self.max_size:
            return bool(self.bits & (1 << element))
        return False
    
    def union(self, other):
        """–û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤"""
        result = BitSetOperations(self.max_size)
        result.bits = self.bits | other.bits
        return result
    
    def intersection(self, other):
        """–ü–µ—Ä–µ—Å–µ—á–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤"""
        result = BitSetOperations(self.max_size)
        result.bits = self.bits & other.bits
        return result
    
    def complement(self):
        """–î–æ–ø–æ–ª–Ω–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤–∞"""
        result = BitSetOperations(self.max_size)
        result.bits = ~self.bits & ((1 << self.max_size) - 1)
        return result
    
    def subset_count(self):
        """–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤ (2^n)"""
        return 1 << bin(self.bits).count('1')

def knapsack_bitmask(weights, values, capacity):
    """–†–µ—à–µ–Ω–∏–µ –∑–∞–¥–∞—á–∏ –æ —Ä—é–∫–∑–∞–∫–µ —á–µ—Ä–µ–∑ –±–∏—Ç–æ–≤—ã–µ –º–∞—Å–∫–∏"""
    n = len(weights)
    max_value = 0
    
    # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ –≤–æ–∑–º–æ–∂–Ω—ã–µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞ (2^n)
    for mask in range(1 << n):
        current_weight = 0
        current_value = 0
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç –≤ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–µ

            if mask & (1 << i):  # –ï—Å–ª–∏ i-–π —ç–ª–µ–º–µ–Ω—Ç –≤–∫–ª—é—á–µ–Ω
                current_weight += weights[i]
                current_value += values[i]
        
        if current_weight <= capacity:
            max_value = max(max_value, current_value)
    
    return max_value

    """–ü—Ä–∏–º–µ—Ä—ã –ø—Ä–∏–º–µ–Ω–µ–Ω–∏—è DP –æ—Å–Ω–æ–≤–∞–Ω–Ω—ã—Ö –Ω–∞ —Ä–µ–∫—É—Ä—Ä–µ–Ω—Ç–Ω—ã—Ö —Å–æ–æ—Ç–Ω–æ—à–µ–Ω–∏—è—Ö"""

        Fibonacci: F(n) = F(n-1) + F(n-2), F(0)=0, F(1)=1
        –í—Ä–µ–º—è: O(n), –ü–∞–º—è—Ç—å: O(n) ‚Üí O(1) —Å –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏–µ–π

        –†–µ–∫—É—Ä—Ä–µ–Ω—Ç–Ω–æ: C(n) = Œ£(i=0 to n-1) C(i) * C(n-1-i)
        –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ: –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–∞–≤–∏–ª—å–Ω—ã—Ö —Å–∫–æ–±–æ—á–Ω—ã—Ö –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–µ–π

            return 1

        dp[0], dp[1] = 1, 1
        
        for i in range(2, n + 1):
            for j in range(i):
                dp[i] += dp[j] * dp[i - 1 - j]
        
        return dp[n]

    def edit_distance(str1, str2):
        """
        –†–∞—Å—Å—Ç–æ—è–Ω–∏–µ –õ–µ–≤–µ–Ω—à—Ç–µ–π–Ω–∞

            DP[i][j-1] + 1,      # insertion

                dp[i][j] = min(
                    dp[i-1][j] + 1,      # deletion
                    dp[i][j-1] + 1,      # insertion
                    dp[i-1][j-1] + cost  # substitution
                )
        
        return dp[m][n]

                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        
        return dp[m][n]

def subset_sum_bitmask(numbers, target):
    """–ü—Ä–æ–≤–µ—Ä–∫–∞ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–∞ —Å –∑–∞–¥–∞–Ω–Ω–æ–π —Å—É–º–º–æ–π"""

def traveling_salesman_bitmask(distances):
    """TSP —á–µ—Ä–µ–∑ –±–∏—Ç–æ–≤—ã–µ –º–∞—Å–∫–∏ –∏ DP"""
    n = len(distances)
    
    # dp[mask][i] = –º–∏–Ω–∏–º–∞–ª—å–Ω–∞—è —Å—Ç–æ–∏–º–æ—Å—Ç—å –ø–æ—Å–µ—Ç–∏—Ç—å –≤—Å–µ –≥–æ—Ä–æ–¥–∞ –≤ mask, –∑–∞–∫–∞–Ω—á–∏–≤–∞—è –≤ i
    dp = [[float('inf')] * n for _ in range(1 << n)]
    dp[1][0] = 0  # –ù–∞—á–∏–Ω–∞–µ–º —Å –≥–æ—Ä–æ–¥–∞ 0
    
    for mask in range(1 << n):
        for u in range(n):
            if not (mask & (1 << u)):
                continue
            
            for v in range(n):
                if mask & (1 << v):
                    continue
                
                new_mask = mask | (1 << v)
                dp[new_mask][v] = min(dp[new_mask][v], 
                                    dp[mask][u] + distances[u][v])
    
    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º—Å—è –≤ –Ω–∞—á–∞–ª—å–Ω—ã–π –≥–æ—Ä–æ–¥
    full_mask = (1 << n) - 1
    min_cost = float('inf')
    for i in range(1, n):
        min_cost = min(min_cost, dp[full_mask][i] + distances[i][0])
    
    return min_cost

def matrix_chain_multiplication(dimensions):
    """
    –û–ø—Ç–∏–º–∞–ª—å–Ω–∞—è —Å–∫–æ–±–æ—á–Ω–∞—è —Ä–∞—Å—Å—Ç–∞–Ω–æ–≤–∫–∞ –ø—Ä–∏ —É–º–Ω–æ–∂–µ–Ω–∏–∏ —Ü–µ–ø–æ—á–∫–∏ –º–∞—Ç—Ä–∏—Ü
    DP[i][j] = –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–∫–∞–ª—è—Ä–Ω—ã—Ö —É–º–Ω–æ–∂–µ–Ω–∏–π –¥–ª—è –º–∞—Ç—Ä–∏—Ü i..j
    """
    n = len(dimensions) - 1
    dp = [[0] * n for _ in range(n)]
    
    # l - –¥–ª–∏–Ω–∞ —Ü–µ–ø–æ—á–∫–∏
    for l in range(2, n + 1):
        for i in range(n - l + 1):
            j = i + l - 1
            dp[i][j] = float('inf')
            
            for k in range(i, j):
                cost = (dp[i][k] + dp[k+1][j] + 
                       dimensions[i] * dimensions[k+1] * dimensions[j+1])
                dp[i][j] = min(dp[i][j], cost)
    
    return dp[0][n-1]

def coin_change_combinations(coins, amount):
    """
    –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø–æ—Å–æ–±–æ–≤ –Ω–∞–±—Ä–∞—Ç—å —Å—É–º–º—É amount –∏—Å–ø–æ–ª—å–∑—É—è –º–æ–Ω–µ—Ç—ã coins
    DP[i] = –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ø–æ—Å–æ–±–æ–≤ –Ω–∞–±—Ä–∞—Ç—å —Å—É–º–º—É i
    """

    dp[0] = 1  # –û–¥–∏–Ω —Å–ø–æ—Å–æ–± –Ω–∞–±—Ä–∞—Ç—å 0 - –Ω–µ –±—Ä–∞—Ç—å –º–æ–Ω–µ—Ç
    
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] += dp[i - coin]
    
    return dp[amount]
