func (o *Order) Apply(event DomainEvent) {
    switch e := event.(type) {
    case OrderCreatedEvent:
        o.id = e.OrderID
        o.status = OrderStatusDraft
    case ItemAddedEvent:
        o.items = append(o.items, e.Item)
    case OrderCompletedEvent:
        o.status = OrderStatusCompleted
    }
}
